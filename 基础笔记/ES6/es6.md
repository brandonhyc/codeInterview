### Destruction 
```js
const [name, age] = ['John', 26];
// name = 'John'; age = 26;

const [name, age] = (() => { return ['Y', 11]})();

const obj = { firstName: 'John', lastName: 'Smith'};
const {firstName, lastName} = obj;
// firstName = 'John', lastName = 'Smith';

const {firstName: a, lastName: b} = obj;
// a = 'John', b = 'Smith'
```
